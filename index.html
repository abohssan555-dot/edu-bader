-- ============================================================ -- تحديث: نظام تفويض الصلاحيات الكامل -- (رئيس الفريق التنفيذي / عضو فريق / مقدم خدمة / مشرف إدارة مدرسية) -- حماية حقيقية على مستوى قاعدة البيانات (RLS + Trigger)، وليست حماية واجهة فقط -- نفّذ هذا الكود في Supabase → SQL Editor → New query → Run -- (يُنفَّذ بأمان حتى لو أُعيد تنفيذه أكثر من مرة) -- ============================================================ -- 1) توسيع الأدوار المسموحة alter table public.profiles drop constraint if exists profiles_role_check; alter table public.profiles add constraint profiles_role_check check (role in ('admin','editor','viewer','provider','team_member','admin_supervisor','team_leader')); -- 2) ربط كل مدرسة بمقدّم الخدمة و/أو عضو الفريق المسؤولين عنها (اختياري لكل منهما) alter table public.schools add column if not exists assigned_provider_id uuid references public.profiles(id); alter table public.schools add column if not exists assigned_team_member_id uuid references public.profiles(id); -- 3) دوال مساعدة create or replace function public.is_my_school(p_school_id uuid) returns boolean as $$ select exists( select 1 from public.schools where id = p_school_id and assigned_provider_id = auth.uid() ); $$ language sql stable security definer; create or replace function public.is_my_team_school(p_school_id uuid) returns boolean as $$ select exists( select 1 from public.schools where id = p_school_id and assigned_team_member_id = auth.uid() ); $$ language sql stable security definer; -- 4) دليل مصغّر بأسماء مقدّمي الخدمة وأعضاء الفريق (لتعبئة قوائم الإسناد) دون كشف كل بيانات المستخدمين drop policy if exists "approved read provider directory" on public.profiles; drop policy if exists "approved read assignable directory" on public.profiles; create policy "approved read assignable directory" on public.profiles for select using (public.is_approved() and role in ('provider','team_member')); -- ============================================================ -- سياسات جدول المدارس (schools) -- ============================================================ drop policy if exists "provider view assigned schools" on public.schools; create policy "provider view assigned schools" on public.schools for select using (public.current_role() = 'provider' and assigned_provider_id = auth.uid()); drop policy if exists "team_member and supervisor view all schools" on public.schools; drop policy if exists "broad roles view all schools" on public.schools; create policy "broad roles view all schools" on public.schools for select using (public.current_role() in ('team_member','admin_supervisor','team_leader')); drop policy if exists "provider update assigned schools" on public.schools; create policy "provider update assigned schools" on public.schools for update using (public.current_role() = 'provider' and assigned_provider_id = auth.uid()); drop policy if exists "team_member update unassigned schools" on public.schools; drop policy if exists "team_member update own assigned schools" on public.schools; create policy "team_member update own assigned schools" on public.schools for update using (public.current_role() = 'team_member' and assigned_team_member_id = auth.uid()); drop policy if exists "admin_supervisor update schools" on public.schools; create policy "admin_supervisor update schools" on public.schools for update using (public.current_role() = 'admin_supervisor'); drop policy if exists "team_leader update school assignments" on public.schools; create policy "team_leader update school assignments" on public.schools for update using (public.current_role() = 'team_leader'); -- حارس: مشرف الإدارة المدرسية يعدّل فقط حقول الكادر ونوع الإدارة، ولا شيء غيرها create or replace function public.guard_admin_supervisor_schools() returns trigger as $$ declare v_role text; begin select role into v_role from public.profiles where id = auth.uid(); if v_role = 'admin_supervisor' then if NEW.name is distinct from OLD.name or NEW.gender is distinct from OLD.gender or NEW.stage is distinct from OLD.stage or NEW.students_count is distinct from OLD.students_count or NEW.teachers_count is distinct from OLD.teachers_count or NEW.principal is distinct from OLD.principal or NEW.supervisor is distinct from OLD.supervisor or NEW.location is distinct from OLD.location or NEW.notes is distinct from OLD.notes or NEW.graduates_1447 is distinct from OLD.graduates_1447 or NEW.expected_graduates_1448 is distinct from OLD.expected_graduates_1448 or NEW.assigned_provider_id is distinct from OLD.assigned_provider_id or NEW.assigned_team_member_id is distinct from OLD.assigned_team_member_id or NEW.archived is distinct from OLD.archived then raise exception 'مشرف الإدارة المدرسية يمكنه تعديل حقول الكادر ونوع الإدارة فقط'; end if; end if; return NEW; end; $$ language plpgsql security definer; drop trigger if exists trg_guard_admin_supervisor_schools on public.schools; create trigger trg_guard_admin_supervisor_schools before update on public.schools for each row execute procedure public.guard_admin_supervisor_schools(); -- حارس: رئيس الفريق التنفيذي يعدّل فقط حقلي إسناد مقدّم الخدمة وعضو الفريق، ولا شيء غيرهما create or replace function public.guard_team_leader_schools() returns trigger as $$ declare v_role text; begin select role into v_role from public.profiles where id = auth.uid(); if v_role = 'team_leader' then if NEW.name is distinct from OLD.name or NEW.gender is distinct from OLD.gender or NEW.stage is distinct from OLD.stage or NEW.students_count is distinct from OLD.students_count or NEW.teachers_count is distinct from OLD.teachers_count or NEW.deputy_count is distinct from OLD.deputy_count or NEW.counselor_count is distinct from OLD.counselor_count or NEW.admin_assistant_count is distinct from OLD.admin_assistant_count or NEW.admin_type is distinct from OLD.admin_type or NEW.count_basis is distinct from OLD.count_basis or NEW.graduates_1447 is distinct from OLD.graduates_1447 or NEW.expected_graduates_1448 is distinct from OLD.expected_graduates_1448 or NEW.principal is distinct from OLD.principal or NEW.supervisor is distinct from OLD.supervisor or NEW.location is distinct from OLD.location or NEW.notes is distinct from OLD.notes or NEW.archived is distinct from OLD.archived then raise exception 'رئيس الفريق التنفيذي يمكنه فقط إسناد المدارس لمقدّم الخدمة أو عضو الفريق، لا تعديل بياناتها'; end if; end if; return NEW; end; $$ language plpgsql security definer; drop trigger if exists trg_guard_team_leader_schools on public.schools; create trigger trg_guard_team_leader_schools before update on public.schools for each row execute procedure public.guard_team_leader_schools(); -- ============================================================ -- سياسات جدول السجلات (records) — الأداء المدرسي، التقويم، وكل الفئات المرتبطة بمدرسة -- ============================================================ drop policy if exists "provider view assigned school records" on public.records; create policy "provider view assigned school records" on public.records for select using (public.current_role() = 'provider' and school_id is not null and public.is_my_school(school_id)); drop policy if exists "team_member and supervisor view all records" on public.records; drop policy if exists "broad roles view all records" on public.records; create policy "broad roles view all records" on public.records for select using (public.current_role() in ('team_member','admin_supervisor','team_leader')); drop policy if exists "provider write assigned school records" on public.records; create policy "provider write assigned school records" on public.records for update using (public.current_role() = 'provider' and school_id is not null and public.is_my_school(school_id)); drop policy if exists "provider insert assigned school records" on public.records; create policy "provider insert assigned school records" on public.records for insert with check (public.current_role() = 'provider' and school_id is not null and public.is_my_school(school_id)); drop policy if exists "team_member write unassigned school records" on public.records; drop policy if exists "team_member write own assigned school records" on public.records; create policy "team_member write own assigned school records" on public.records for update using (public.current_role() = 'team_member' and school_id is not null and public.is_my_team_school(school_id)); drop policy if exists "team_member insert unassigned school records" on public.records; drop policy if exists "team_member insert own assigned school records" on public.records; create policy "team_member insert own assigned school records" on public.records for insert with check (public.current_role() = 'team_member' and school_id is not null and public.is_my_team_school(school_id)); drop policy if exists "admin_supervisor update performance records" on public.records; create policy "admin_supervisor update performance records" on public.records for update using (public.current_role() = 'admin_supervisor' and category = 'school_performance'); -- حارس: مشرف الإدارة المدرسية يعدّل فقط "نسبة مجال الإدارة المدرسية" داخل الأداء المدرسي، ولا شيء غيره create or replace function public.guard_admin_supervisor_records() returns trigger as $$ declare v_role text; old_extra jsonb; new_extra jsonb; begin select role into v_role from public.profiles where id = auth.uid(); if v_role = 'admin_supervisor' then if OLD.category is distinct from 'school_performance' then raise exception 'لا يملك هذا الدور صلاحية تعديل هذه الفئة'; end if; if NEW.category is distinct from OLD.category or NEW.title is distinct from OLD.title or NEW.name is distinct from OLD.name or NEW.school_id is distinct from OLD.school_id or NEW.date is distinct from OLD.date or NEW.status is distinct from OLD.status or NEW.priority is distinct from OLD.priority or NEW.description is distinct from OLD.description or NEW.notes is distinct from OLD.notes or NEW.link is distinct from OLD.link or NEW.archived is distinct from OLD.archived then raise exception 'مشرف الإدارة المدرسية يمكنه تعديل نسبة مجال الإدارة المدرسية فقط'; end if; old_extra := coalesce(OLD.extra, '{}'::jsonb) - 'adminDomainPct'; new_extra := coalesce(NEW.extra, '{}'::jsonb) - 'adminDomainPct'; if old_extra is distinct from new_extra then raise exception 'مشرف الإدارة المدرسية يمكنه تعديل نسبة مجال الإدارة المدرسية فقط'; end if; end if; return NEW; end; $$ language plpgsql security definer; drop trigger if exists trg_guard_admin_supervisor_records on public.records; create trigger trg_guard_admin_supervisor_records before update on public.records for each row execute procedure public.guard_admin_supervisor_records();
